CREATE TABLE TabelaA(ID NUMBER NOT NULL,
                     NAZIV VARCHAR2(100),
                     DATUM DATE,
                     CIJELIBROJ NUMBER,
                     REALNIBROJ NUMBER(*,2),
                     CONSTRAINT tabA_cijeliBroj_ck CHECK(CIJELIBROJ NOT BETWEEN 5 AND 15),
                     CONSTRAINT tabA_realniBroj_ck CHECK(REALNIBROJ > 5 ),
                     CONSTRAINT tabA_id_pk PRIMARY KEY(ID)
                     );
CREATE TABLE TabelaB(ID NUMBER NOT NULL,
                     NAZIV VARCHAR2(100),
                     DATUM DATE,
                     CIJELIBROJ NUMBER,
                     REALNIBROJ NUMBER(*,2),
                     FKTabelaA NUMBER NOT NULL,
                     CONSTRAINT tabB_fkTabelaB_fk FOREIGN KEY(FKTabelaA)
                     REFERENCES TabelaA(ID),
                     CONSTRAINT tabB_cijeliBroj_uk UNIQUE(CIJELIBROJ),
                     CONSTRAINT tabB_id_pk PRIMARY KEY(ID));
